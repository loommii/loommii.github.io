package main

import (
       "runtime"
       "sync"
)

func main() {
       runtime.GOMAXPROCS(1)

       var wg sync.WaitGroup
       wg.Add(3)
       go func(n int) {
              println(n)
              wg.Done()
       }(1)
       go func(n int) {
              println(n)
              wg.Done()
       }(2)
       go func(n int) {
              println(n)
              wg.Done()
       }(3)
       wg.Wait()
}
