func knightDialer(n int) (ans int) {
    mod := 1000000000 + 7
    memo := make([][]int, 10) // memo[y][n]
    for i := 0; i < 10; i++ {
        memo[i] = make([]int, n)
    }
    var dfs func(int, int) int
    dfs = func(y, n int) int {
        if n == 0 {
            return 1
        }
        ans := 0
        for _, v := range dirs[y] {
            if memo[v][n-1] != 0 {
                ans += memo[v][n-1]
            } else {
                ans += dfs(v, n-1)
            }
        }
        ans %= mod
        memo[y][n] = ans
        return ans
    }
    for i := 0; i < 10; i++ {
        ans += dfs(i, n-1)
    }
    return ans % mod
}
