func findSpecialInteger(arr []int) (ans int) {
    n := len(arr) / 4
    f := arr[0]
    i := 0
    for _, v := range arr {
        if v == f {
            i++
        } else {
            f = v
            i = 1
        }
        if i > n {
            ans = f
            break
        }
    }
    return
}
