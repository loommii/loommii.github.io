func semiOrderedPermutation(nums []int) int {
    j, k := 0, 0
    n := len(nums)
    for i, v := range nums {
        if v == 1 {
            j = i
        }
        if v == n {
            k = i
        }
    }
    ans := j
    ans += (n - 1 - k)
    if j > k {
        ans--
    }
    return ans
}
